자바스크립트 중급 강좌 #7 - 배열 메소드1(Array methods)

Array

  1. push() : 뒤에 삽입
  1. pop() : 뒤에 삭제
  1. unshift() : 앞에 삽입
  1. shift() : 앞에 삭제
  1. splice(n,m) : 특정 요소 지움. n : 시작, m : 개수
    let arr = [1,2,3,4,5];
arr.splice(1,2);

console.log(arr);
> [1, 4, 5]
  1. splice(n,m,x) : 특정 요소 지우고, 추가. m에 0을 넣으면, 아무것도 지워지지 않음.
    arr.splice(1,2,100);

console.log(arr);
> [1, 100]
  1. splice() : 삭제된 요소 반환.
    let arr = [1,2,3,4,5];
let result = arr.splice(1,2);

console.log(arr);
> [1, 4, 5]
console.log(result);
> [2, 3]
  1. slice(n,m) : n부터 m까지 반환. slice()면, 배열에 있는 값 모두 반환 ( == 배열이 그대로 복사됨.)
  1. concat(arr2, arrr3, …) : 합쳐서 새 배열을 반환.
    let arr1 = [1,2];
arr1.concat([3,4]);
> [1, 2, 3, 4]
  1. forEach(fn) : 배열 반환.
    let users = ['Mike', 'Tom', 'Jane'];
users.forEach((name, index) => {
    console.log(name);
});
> Mike
> Tom
> Jane
    let users = ['Mike', 'Tom', 'Jane'];
users.forEach((name, index) => {
    console.log(`${index + 1}, ${name}`);
});
> 1, Mike
> 2, Tom
> 3, Jane
  1. indexOf / lastIndexOf : 문자열의 인덱스 반환. 없으면 -1 반환.
  1. includes() : 포함하는지 확인. 있으면 true, 없으면 false.
  1. find(fn), findIndex(fn) : 인덱스를 찾음. 함수형태로. 첫번째 true값만 반환하고 끝, 만약 없으면 undefined를 반환함. 
    let arr = [1,2,3,4,5];
const result1 = arr.find((item) => {
    return item % 2 === 0;
});

console.log(result1);
> 2
    let userList = [
    {name:"mike", age:30},
    {name:"Jane", age:27},
    {name:"Tom", age:10},
    ];
const result2 = userList.find((user) => {
    if (user.age < 19){
        return true;
    }else{
        return false;
    }
});

console.log(result2);
> {name: 'Tom', age: 10}


const result2 = userList.findIndex((user) => {
    if (user.age < 19){
        return true;
    }else{
        return false;
    }
});
console.log(result2);
> 2
  1. filter(fn) : 만족하는 모든 요소를 배열로 반환. 
    let arr = [1,2,3,4,5,6];
const result3 = arr.filter((item) => {
    return item % 2=== 0;
});

console.log(result3);
> [2, 4, 6]
  1. reverse() : 역순으로 재정렬
  1. map(fn) : 함수를 받아 특정 기능을 실행하고, 새로운 배열을 반환
    let userList = [
    {name:"mike", age:30},
    {name:"Jane", age:27},
    {name:"Tom", age:10},
];
let newUserList = userList.map((user, index) => {
    return Object.assign({}, user,{
       isAdult: user.age > 19, 
    });
});

console.log(newUserList);
0: {name: 'mike', age: 30, isAdult: true}
1: {name: 'Jane', age: 27, isAdult: true}
2: {name: 'Tom', age: 10, isAdult: false}

///////////////////////////////////////////////////////

let newUserList = userList.map((user, index) => {
    return Object.assign({}, user,{
       id: index + 1,
        isAdult: user.age > 19, 
    });
});

console.log(newUserList);
0: {name: 'mike', age: 30, id: 1, isAdult: true}
1: {name: 'Jane', age: 27, id: 2, isAdult: true}
2: {name: 'Tom', age: 10, id: 3, isAdult: false}
  1. join() : 배열을 합쳐 문자열을 생성
    let arr = ['안녕','나는','진희야'];
let result = arr.join();
console.log(result);
> 안녕,나는,진희야

let result = arr.join(" ");
console.log(result);
> 안녕 나는 진희야
  1. split() : 문자열을 나눠 배열로 생성
    const users2 = "Mike,Jane,Tom,Tony";
const result5 = users2.split(',');
console.log(result5);
> ['Mike', 'Jane', 'Tom', 'Tony']
    const user2 = "Hello, My name is jinny";
const result6 = user2.split("");

> ['H', 'e', 'l', 'l', 'o', ',', ' ', 'M', 'y', ' ', 'n', 'a', 'm', 'e', ' ', 'i', 's', ' ', 'j', 'i', 'n', 'n', 'y']
  1. isArray() : 배열인지 아닌지 판단
    let user = { name: 'mike',age: 30,};
let userList = ["mike","tom","Jane"];
console.log(Array.isArray(user));
> false
console.log(Array.isArray(userList));
> true
